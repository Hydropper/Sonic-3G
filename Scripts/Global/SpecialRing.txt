event ObjectMain
	if object.state == 0
		if object[0].value0 > -1
			BoxCollisionTest(C_TOUCH, object.entityPos, -16, -16, 16, 16, 0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == 1
				PlaySfx(SfxName[Special Ring], 1)
				object.value39 = 1
				object[0].type = TypeName[Blank Object]
				object[playerCount].type = TypeName[Blank Object]
				object[0].controlMode = -1
				object[0].direction = FLIP_NONE
				ResetObjectEntity(4, TypeName[Blank Object], 0, 0, 0)
				stage.activeList = SPECIAL_STAGE
				stage.listPos = 1
				object.state = 1
				object.priority = PRIORITY_ACTIVE_PAUSED
				object[0].controlMode = -1
				fadeColor = 208
				fadeColor <<= 16
				temp0 = 255
				temp0 <<= 8
				fadeColor += temp0
				fadeColor += 224
				object.value9 = 0
			end if
		end if
	end if
	if object.value39 == 1
		object.value40++
		object.value9 += 8
		if object.value40 >= 120
			SetScreenFade(208, 255, 224, object.value9)
			LoadStage()
		end if
	end if
	if PlayerObject_SuperState == 1
		ResetObjectEntity(object.entityPos, TypeName[Blank Object], 0, 0, 0)
	end if
end event


event ObjectDraw
	if object.value41 == 0
		temp1 = object.value38
		temp1 >>= 3
		DrawSprite(temp1, object.xpos, object.ypos)
		object.value38++
		if object.value38 > 31
			object.value38 = 0
		end if
	end if
	switch object.state
	case 0
		break
	case 1
		if object.value31 == 0
			object.frame = 32
			object.value31 = 1
		end if
		temp0 = object.frame
		temp0 >>= 3
		if temp0 > 7
			object.value41 = 1
			object.direction = FLIP_X
		else
			object.direction = FLIP_NONE
		end if
		DrawSpriteFX(temp0, FX_FLIP, object.xpos, object.ypos)
		if temp0 > 10
			object.frame = 88
		end if
		object.frame += 6
		break
	end switch
end event
