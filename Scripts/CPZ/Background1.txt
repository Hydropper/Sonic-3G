// Static Values
public value StartHeight = 0;
public value NewHeight = 0;
public value Height = 0;
public value YFactor = 0;
public value Xleft = 0;
public value Xright = 0;
public value offset = 0;
public value WaterLevelApparent = 0;
public value Min = 0;


// Function declarations

event ObjectMain

	// Defining X right frame and X left frame
	Xleft = screen.xoffset
	Xright = Xleft
	Xright += screen.xsize


	YFactor = 62586

	offset = Xleft
	offset <<= 16
	offset *= 32768

	object.xpos = Xleft

	hud.value1 = screen.yoffset
	hud.value2 = Xright

	if NewHeight >= 0
		object.ypos = WaterLevelApparent
		NewHeight = screen.yoffset
		NewHeight <<= 16
		NewHeight *= YFactor

		if NewHeight < StartHeight
			Min = NewHeight
		else
			Min = StartHeight
		end if

		Height = Min





	end if



end event


event ObjectDraw
	DrawSprite(97)
end event
